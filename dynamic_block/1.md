resource "aws_security_group" "sg" {
  for_each = var.security_groups
  vpc_id = aws_vpc.vpc.id
  name = each.key

  dynamic "ingress" {
    for_each = each.value.inbound_rules
    content {
      from_port   = ingress.value.start_port
      to_port     = try(ingress.value.end_port,ingress.value.start_port)
      protocol    = try(ingress.value.protocol,"tcp")
      description = try(ingress.value.description,null)
      cidr_blocks = try(ingress.value.cidr_blocks,["0.0.0.0/0"])
    }
  }

  dynamic "egress" {
    for_each = each.value.outbound_rules
    content {
      from_port   = egress.value.start_port
      to_port     = try(egress.value.end_port,egress.value.start_port)
      protocol    = try(egress.value.protocol,"tcp")
      description = try(egress.value.description,null)
      cidr_blocks = try(egress.value.cidr_blocks,["0.0.0.0/0"])
    }
  }
}
